<script>

const quotes = [
"Bugün çalıştığın konu yarınki başarındır.",
"Disiplin seni DUS'a götürür.",
"Her soru seni uzmanlığa yaklaştırır.",
"Vazgeçmeyenler kazanır.",
"Rakiplerin dinlenirken sen çalış."
];

document.getElementById("quote").innerText =
quotes[new Date().getDate() % quotes.length];

const topicList = [
"Anatomi - Baş Boyun",
"Anatomi - Nöroanatomi",
"Fizyoloji",
"Biyokimya",
"Mikrobiyoloji",
"Patoloji",
"Farmakoloji",
"Ağız Çene Cerrahisi",
"Periodontoloji",
"Protetik",
"Restoratif",
"Endodonti",
"Ortodonti",
"Pedodonti"
];

let saved = JSON.parse(localStorage.getItem("dusTopics")) || [];

const container = document.getElementById("topics");

topicList.forEach((t,i)=>{
let checked = saved.includes(i);
container.innerHTML +=
`<div>
<input type="checkbox" ${checked?"checked":""}
onclick="toggle(${i})"> ${t}
</div>`;
});

function toggle(i){
if(saved.includes(i)){
saved = saved.filter(x=>x!=i);
}else{
saved.push(i);
}
localStorage.setItem("dusTopics",JSON.stringify(saved));
updateTopicProgress();
}

function updateTopicProgress(){
let percent = Math.round((saved.length/topicList.length)*100);
document.getElementById("tbar").style.width=percent+"%";
document.getElementById("tpercent").innerText="%"+percent+" tamamlandı";
}

function updateQuestion(){
let target = document.getElementById("target").value;
let solved = document.getElementById("solved").value;
let percent = Math.min(Math.round((solved/target)*100),100);
document.getElementById("qbar").style.width=percent+"%";
document.getElementById("qpercent").innerText="%"+percent+" günlük ilerleme";
}

updateTopicProgress();

</script>
